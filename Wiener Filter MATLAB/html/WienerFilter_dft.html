% y: Degraded image (with blur and noise)
% h: Degrade kernel
% sigma: standard deviation of noise

C = size(y,2); %get number of columns

Syy = abs(Yf).^2/(R*C);
See=sigma^2;
Sxx=(Syy-See);

RestoredImage = real(idft2(eXf));
